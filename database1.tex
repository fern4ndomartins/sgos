\newcommand{\dbtable}[3]{
    % #1 = node name, #2 = position, #3 = content
    \node[table] (#1) #2 {
        \begin{tabular}{@{}l@{}}
        \rowcolor{headercolor}
        \textcolor{white}{\textbf{#3}}
        \end{tabular}
    };
}
